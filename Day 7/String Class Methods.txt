4. charAt()
	String str = "Hello";

5. indexOf()
	String str = "Hello";
	int i = str.indexOf('e');
	s.o.p(i);		3
	
6. lastIndexOf()
	int i = str.lastIndexOf('l');
	s.o.p(i);		3

NOTE : If character not found, -1 is returned. Also, it is Case Sensitive.

7. substring()
	String s = "Good Evening";
	String s1, s2;
	s1 = s.substring(5,8);	Eve		5 is included and 8 is not included
	s2 = s.substring(5);	Evening
